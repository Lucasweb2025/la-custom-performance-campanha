      combo_care: { id: 'combo_care', nome: 'Combo Care', categoria: 'Premium', pontos: 30, desc: 'Pacote completo de proteção', tipo: 'premium' },
      parabrisa: { id: 'parabrisa', nome: 'Para-brisa', categoria: 'Estratégico', pontos: 20, desc: 'Película protetora de para-brisa', tipo: 'estrategico' },

      higienizacao: { id: 'higienizacao', nome: 'Higienização', categoria: 'Constância', pontos: 6, desc: 'Limpeza técnica interna', tipo: 'constancia' },

    };

    const COMBOS_REGRAS = [
      { id: 'ouro', nome: 'Combo Ouro L.A', itens: ['combo_care', 'crystaline', 'ppf_full'], multiplicador: 2.0, tag: 'Multiplicador Máximo' },
      { id: 'total', nome: 'Proteção Total', itens: ['ppf_full', 'parabrisa', 'sas'], multiplicador: 1.8, tag: 'Rota Black' },
      { id: 'blindagem', nome: 'Blindagem Premium', itens: ['ppf_full', 'parabrisa'], multiplicador: 1.5, tag: 'Acelerador Gold' }

      const [isLoggedIn, setIsLoggedIn] = useState(false);
      const [currentTab, setCurrentTab] = useState('inicio'); 
      
      const [vendedores, setVendedores] = useState([
        { id: 'ricardo', nome: 'Ricardo Mendes', concessionaria: 'BMW Morumbi', vendas: 18, pontos: 248 },
        { id: 'camila', nome: 'Camila Tavares', concessionaria: 'Audi Berrini', vendas: 14, pontos: 196 },
        { id: 'felipe', nome: 'Felipe Andrade', concessionaria: 'Porsche Centro SP', vendas: 12, pontos: 172, isMe: true },
        { id: 'juliana', nome: 'Juliana Costa', concessionaria: 'BMW Morumbi', vendas: 11, pontos: 138 },
        { id: 'marcos', nome: 'Marcos Vinícius', concessionaria: 'Mercedes JK', vendas: 10, pontos: 124 },
        { id: 'priscila', nome: 'Priscila Ramos', concessionaria: 'Audi Berrini', vendas: 9, pontos: 96 },

        { id: 1, chassi: 'WBA11X-902105', vendedorId: 'ricardo', vendedorNome: 'Ricardo Mendes', produtos: 'PPF Full + Para-brisa', pontos: 120, valor: 8500, status: 'Pendente' },
        { id: 2, chassi: 'WP0AC2-884212', vendedorId: 'camila', vendedorNome: 'Camila Tavares', produtos: 'Crystalline + SAS 3M', pontos: 60, valor: 6800, status: 'Pendente' }

      const [formVendedorId, setFormVendedorId] = useState('felipe');

      const [cadNome, setCadNome] = useState('');

        setFormProdutos(prev => prev.includes(id) ? prev.filter(p => p !== id) : [...prev, id]);
      };

      const handleSubmeterVenda = (e) => {
        e.preventDefault();
        if (!formChassi) return alert('Por favor, digite o Chassi ou número do contrato.');
        if (formProdutos.length === 0) return alert('Selecione pelo menos um produto para lançar.');

        let maxMult = 1.0;
        COMBOS_REGRAS.forEach(combo => {
          if (combo.itens.every(item => formProdutos.includes(item)) && combo.multiplicador > maxMult) {
            maxMult = combo.multiplicador;
          }
        });

          chassi: formChassi.toUpperCase(),

          vendedorNome: vendedorObj.nome,
          produtos: formProdutos.map(p => PRODUTOS_DB[p].nome).join(' + '),
          pontos: Math.round(basePoints * maxMult),
          valor: formProdutos.length * 4200, 
          status: 'Pendente'

        setFormChassi('');
        setFormProdutos([]);

        setCurrentTab('gerente'); // Redirecionamento reativo corrigido

        e.preventDefault();
        if (!cadNome) return alert('Por favor, digite o nome completo.');

          return alert('❌ ACESSO NEGADO: Chave Corporativa incorreta.');

          id: novoId,
          nome: cadNome,
          concessionaria: cadLoja,
          vendas: 0,
          pontos: 0,
          isMe: true 
        };

        alert(`🎯 Credenciamento Aceito! Bem-vindo, ${cadNome}.`);
        setIsLoggedIn(true);
        setCurrentTab('inicio');

        setContratos(prev => prev.map(c => c.id === contrato.id ? { ...c, status: 'Validado' } : c));

          if (v.id === contrato.vendedorId) {
            return { ...v, vendas: v.vendas + 1, pontos: v.pontos + contrato.pontos };
          }
          return v;
        }));
      };

        setContratos(prev => prev.map(c => c.id === id ? { ...c, status: 'Recusado' } : c));

        let pendentes = 0;
        contratos.forEach(c => {
          if (c.status === 'Validado') faturamentoGlobal += c.valor;
          if (c.status === 'Pendente') pendentes += 1;

          let vendasFinais = v.vendas;
          if (v.isMe) {
            pontosFinais += simTotalVenda;
            if (simTotalVenda > 0) vendasFinais += 1;
          }
          
          let tier = 'INICIANTE';
          if (pontosFinais >= 220) tier = 'BLACK';
          else if (pontosFinais >= 140) tier = 'GOLD';
          else if (pontosFinais >= 80) tier = 'SILVER';
          else if (pontosFinais >= 40) tier = 'BRONZE';

            lojasAgrupadas[v.concessionaria].vendas += v.vendas;

            const vObj = vendedores.find(v => v.id === c.vendedorId);

              lojasAgrupadas[vObj.concessionaria].baseFaturamento += c.valor;
            }
          }
        });

        return { simBase, simMaxMult, simTotalVenda, faturamentoGlobal, pendentes, rankingOrdenado, dashboardLojas, simNivel, simCor, simPremios };

        if (authMode === 'login') {
          return (
            <div className="min-h-screen flex items-center justify-center p-4 bg-[#08090b]">
              <div className="bg-[#111317] border border-[#1c1f26] p-8 rounded-md max-w-sm w-full shadow-2xl transition-all hover:border-[#e5a93b]/30">
                <div className="flex flex-col items-center mb-6">
                  <div className="bg-[#e5a93b] text-black font-black px-3 py-1.5 rounded text-xl tracking-tighter mb-3">LA</div>

                  <div>

                    <input type="text" defaultValue="performance.la" className="w-full bg-[#08090b] border border-[#1c1f26] rounded p-2.5 text-sm text-white focus:outline-none focus:border-[#e5a93b]" />

                  </div>

                    Entrar no Sistema
                  </button>
                  <div className="text-center pt-2">
                    <span onClick={() => setAuthMode('cadastro')} className="text-xs text-[#868c96] hover:text-[#e5a93b] cursor-pointer underline">
                      É um Executivo Novo? Cadastre-se aqui

                    <label className="block text-[10px] font-bold uppercase text-[#868c96] mb-1.5 tracking-wider">Sua Concessionária</label>

                    Concluir Credenciamento
                  </button>
                  <div className="text-center pt-2">
                    <span onClick={() => setAuthMode('login')} className="text-xs text-[#868c96] hover:text-white cursor-pointer underline">
                      Voltar para o Login padrão

                <i className="fa-solid fa-house mr-1.5"></i>Início Geral
              </button>

                <i className="fa-solid fa-square-plus mr-1.5"></i>Lançar Venda
              </button>
              <button onClick={() => setCurrentTab('gerente')} className={`px-4 py-2 rounded-md transition-all flex items-center gap-1.5 whitespace-nowrap ${currentTab === 'gerente' ? 'bg-[#e5a93b] text-black' : 'text-[#868c96] hover:text-white'}`}>
                <i className="fa-solid fa-user-gear"></i>Painel do Gerente
                {dadosCalculados.pendentes > 0 && <span className="bg-rose-500 text-white font-sans text-[9px] px-1.5 py-0.5 rounded-full animate-pulse">{dadosCalculados.pendentes}</span>}
              </button>
            </nav>

            <div className="flex items-center gap-3 self-end lg:self-center">
              <button 
                onClick={handleLogout}
                className="text-xs text-rose-400 hover:text-rose-500 font-bold flex items-center gap-1 bg-rose-500/5 border border-rose-500/10 px-3 py-1.5 rounded transition-all hover:bg-rose-500/10 active:scale-95"
              >
                <i className="fa-solid fa-right-from-bracket"></i> Sair
              </button>
            </div>
          </header>

          {/* TELA 1: INÍCIO GERAL COMPLETA */}
          {currentTab === 'inicio' && (
            <div className="space-y-6">
              <section className="bg-[#111317] border border-[#1c1f26] p-6 rounded-md">
                <div className="text-[10px] uppercase tracking-widest text-[#e5a93b] font-bold flex items-center gap-1.5 mb-1"><span>♦</span> CAMPANHA PREMIUM</div>
                <h2 className="text-3xl font-black font-mono uppercase">PROGRAMA PERFORMANCE PREMIUM</h2>
                <p className="text-[#868c96] text-xs max-w-2xl mt-2 leading-relaxed">Direcionamento estratégico para alta rentabilidade. Use o simulador tátil abaixo e veja a reordenação instantânea do Leaderboard.</p>
              </section>

              {/* CARDS KPI DE NEGÓCIO */}

                <div className="bg-[#111317] border border-[#1c1f26] p-4 rounded-md">

                  <h4 className="text-xl font-black font-mono text-emerald-400 mt-0.5">R$ {dadosCalculados.faturamentoGlobal.toLocaleString('pt-BR')}</h4>
                </div>
                <div className="bg-[#111317] border border-[#1c1f26] p-4 rounded-md">
                  <span className="text-[9px] text-[#868c96] uppercase font-bold tracking-wider">Ticket Médio F&I</span>
                  <h4 className="text-xl font-black font-mono text-[#38bdf8] mt-0.5">R$ 8.450 <span className="text-xs font-normal font-sans text-zinc-500">/ veículo</span></h4>
                </div>
                <div className="bg-[#111317] border border-[#1c1f26] p-4 rounded-md">
